📝 快速开始

// 创建 New.js 应用
const New = require('newjs');

const app = new New();

// 首页路由
app.get('/', (req, res) => {
    res.render('index.html', {
        title: 'New.js 应用',
        message: '欢迎使用 New.js 框架'
    });
});

// 支付宝支付示例
app.post('/api/payment', async (req, res) => {
    const result = await app.alipay.createOrder({
        orderId: 'ORDER_' + Date.now(),
        amount: 100.00,
        subject: '商品名称'
    });
    res.json({ payUrl: result.payUrl });
});

// 页面跳转
app.get('/redirect', (req, res) => {
    res.redirect('/');
});

// 启动服务器
app.listen(3000, () => {
    console.log('Server running on http://localhost:3000');
});
