🚀 介绍

New.js 是一个现代化、高性能的 Node.js 后端框架，专为构建高效、可扩展的 Web 应用而设计。它提供了丰富的功能和简洁的 API，让开发者能够快速构建生产级别的应用。

✨ 核心特性

  • ⚡ 高性能 - 基于原生 Node.js，支持集群模式
  • 💰 支付宝支付 - 完整集成支付宝支付接口
  • 🔄 页面跳转 - 支持服务端重定向和客户端路由
  • 💾 数据库支持 - 内置 SQLite，支持 MySQL、PostgreSQL
  • 🔌 WebSocket - 原生 WebSocket 支持，实时通信
  • 🛡️ 安全防护 - CSRF、XSS、速率限制等安全机制
  • 📝 .njs 文件支持 - 专用后端文件格式

📝 第一个应用

创建一个简单的 Hello World 应用：

// app/main.njs
const New = require('newjs');

const app = new New();

// 首页路由
app.get('/', (req, res) => {
    res.send('Hello New.js!');
});

// API路由
app.get('/api/hello', (req, res) => {
    res.json({
        message: 'Hello from New.js!',
        timestamp: Date.now()
    });
});

// 启动服务器
app.listen(3000, () => {
    console.log('Server running on http://localhost:3000');
});

🛣️ 路由

New.js 提供了灵活的路由系统，支持 GET、POST、PUT、DELETE 等 HTTP 方法。

基本路由

// GET 请求
app.get('/users', (req, res) => {
    res.json({ users: [] });
});

// POST 请求
app.post('/users', (req, res) => {
    const user = req.body;
    res.json({ user, created: true });
});

// PUT 请求
app.put('/users/:id', (req, res) => {
    const { id } = req.params;
    const data = req.body;
    res.json({ id, data });
});

// DELETE 请求
app.delete('/users/:id', (req, res) => {
    const { id } = req.params;
    res.json({ deleted: true });
});

路由参数

// 参数路由
app.get('/users/:userId/posts/:postId', (req, res) => {
    const { userId, postId } = req.params;
    res.json({ userId, postId });
});

路由分组

app.group('/api', (router) => {
    router.get('/users', (req, res) => {
        res.json({ users: [] });
    });
    
    router.post('/users', (req, res) => {
        res.json({ created: true });
    });
});

🔧 中间件

中间件函数可以访问请求对象和响应对象，并可以修改它们或终止请求。

使用中间件

// 日志中间件
app.use((req, res, next) => {
    console.log(`${req.method} ${req.url}`);
    next();
});

// 认证中间件
app.use((req, res, next) => {
    const token = req.headers.authorization;
    if (!token) {
        res.status(401).json({ error: 'Unauthorized' });
        return;
    }
    next();
});

路由级中间件

const authMiddleware = (req, res, next) => {
    if (req.session.user) {
        next();
    } else {
        res.redirect('/login');
    }
};

app.get('/dashboard', authMiddleware, (req, res) => {
    res.render('dashboard.html');
});

💰 支付宝支付

New.js 内置了完整的支付宝支付集成，支持即时到账、扫码支付等功能。

配置支付宝

// 配置支付宝参数
app.alipay.init({
    appId: 'your-app-id',
    privateKey: 'your-private-key',
    publicKey: 'alipay-public-key',
    notifyUrl: 'https://yourdomain.com/api/alipay/notify',
    returnUrl: 'https://yourdomain.com/payment/success'
});

创建支付订单

app.post('/api/payment/create', async (req, res) => {
    const { amount, subject } = req.body;
    
    const result = await app.alipay.createOrder({
        orderId: 'ORDER_' + Date.now(),
        amount: amount,
        subject: subject,
        body: '商品描述'
    });
    
    res.json({
        success: true,
        payUrl: result.payUrl
    });
});

处理支付回调

app.post('/api/alipay/notify', async (req, res) => {
    const verified = app.alipay.verifyNotify(req.body);

🔌 WebSocket

New.js 内置 WebSocket 支持，实现实时双向通信。

// WebSocket 连接处理
app.websocket.onConnection((ws, req) => {
    console.log('新客户端连接');
    
    ws.on('message', (message) => {
        console.log('收到消息:', message);
        // 广播消息
        app.websocket.broadcast(message);
    });
    
    ws.send('连接成功！');
});

// 发送消息到所有客户端
app.websocket.broadcast('系统通知：服务器重启');

💾 数据库操作

New.js 内置 SQLite 支持，并提供 ORM 模型。

原始查询

// 查询数据
const users = await app.db.query('SELECT * FROM users WHERE age > ?', [18]);

// 执行更新
await app.db.execute('UPDATE users SET status = ? WHERE id = ?', ['active', 1]);

使用模型

// 定义模型
const User = app.db.model('users', {
    username: 'string',
    email: 'string',
    age: 'number'
});

// 创建用户
const userId = await User.create({
    username: 'john',
    email: 'john@example.com',
    age: 25
});

// 查询用户
const user = await User.find(userId);
const activeUsers = await User.where({ status: 'active' });

🔐 会话管理

New.js 提供内置的会话管理功能。

// 设置会话
app.post('/login', (req, res) => {
    req.session.user = {
        id: 1,
        username: 'john'
    };
    res.json({ success: true });
});

// 读取会话
app.get('/profile', (req, res) => {
    const user = req.session.user;
    if (user) {
        res.json(user);
    } else {
        res.status(401).json({ error: '未登录' });
    }
});

// 销毁会话
app.get('/logout', (req, res) => {
    req.session.destroy();
    res.redirect('/');
});

📨 请求和响应

Request 对象

app.get('/example', (req, res) => {
    // 查询参数
    const { page, limit } = req.query;
    
    // 路由参数
    const { id } = req.params;
    
    // 请求体
    const data = req.body;
    
    // 请求头
    const token = req.headers.authorization;
    
    // Cookies
    const sessionId = req.cookies.sessionId;
});

Response 对象

app.get('/response', (req, res) => {
    // JSON 响应
    res.json({ message: 'Hello' });
    
    // HTML 响应
    res.html('<h1>Hello</h1>');
    
    // 重定向
    res.redirect('/home');
    
    // 设置 Cookie
    res.setCookie('token', 'abc123', { maxAge: 3600 });
    
    // 发送文件
    res.sendFile('/path/to/file.pdf');
    
    // 下载文件
    res.download('/path/to/file.pdf', 'filename.pdf');
});

🚢 生产环境部署

使用 PM2 部署

# 安装 PM2
npm install -g pm2

# 启动应用
pm2 start app/main.njs --name newjs-app

# 设置开机自启
pm2 startup
pm2 save

使用 Docker 部署

# Dockerfile
FROM node:18-alpine
WORKDIR /app
COPY package*.json ./
RUN npm install --production
COPY . .
EXPOSE 3000
CMD ["node", "app/main.njs"]
